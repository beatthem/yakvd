def init!(config)
  filename = config[:filename]
  File.open(filename, "r") do |infile|
    line = infile.gets
    # p line
    counter = 0
    while line
      puts "#{counter}: #{line}"
      self.store(line)
      counter = counter + 1
      line = infile.gets
    end
  end
end
