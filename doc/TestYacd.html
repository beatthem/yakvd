def _test_template(
  query_hash,
  allowed_response,
  error_text,
  inverse = false
)
  s = TCPSocket.new('127.0.0.1', 2000)

  s.print("test\r\n")
  s.print("\r\n")
  s.print(
    query_hash.to_json + "\r\n"
  )
  response = s.gets("\r\n")
  if inverse
    assert_not_equal(allowed_response, response, error_text)
  else
    assert_equal(allowed_response, response, error_text)
  end

  s.close
end
